 23class SpatialMap:
 24    """
 25    A simple spatial map to keep track of occupied areas in a container.
 26
 27    **Object Attributes:**
 28    - `width` The width of the container.
 29    - `height` The height of the container.
 30    - `occupied` A list of dictionaries representing occupied areas in the container.
 31    """
 32    def __init__(self, width, height):
 33        self.width = width
 34        self.height = height
 35        self.occupied = []
 36
 37    def is_space_available(self, x, y, w, h):
 38        """
 39        **Params:**
 40        - `x` The x-coordinate of the space.
 41        - `y` The y-coordinate of the space.
 42        - `w` The width of the space.
 43        - `h` The height of the space.
 44
 45        **Returns:**
 46        - `bool` True if the space is available, False otherwise.
 47
 48        Check if the given space is available for a widget.
 49        """
 50        # Check if within container boundaries
 51        if x + w > self.width or y + h > self.height:
 52            return False
 53
 54        # Check for overlap with other widgets
 55        for oc in self.occupied:
 56            if not (x + w <= oc['x'] or x >= oc['x'] + oc['width'] or y + h <= oc['y'] or y >= oc['y'] + oc['height']):
 57                return False
 58
 59        return True
 60
 61    def occupy_space(self, x, y, w, h):
 62        """
 63        **Params:**
 64        - `x` The x-coordinate of the space.
 65        - `y` The y-coordinate of the space.
 66        - `w` The width of the space.
 67        - `h` The height of the space.
 68
 69        Mark the given space as occupied, by adding it to the list of occupied areas.
 70        """
 71        self.occupied.append({'x': x, 'y': y, 'width': w, 'height': h})

 73def place_widget(container: lv.obj, widget: lv.obj, spatial_map: SpatialMap):
 74    """
 75    **Params:**
 76    - `container` The container to place the widget in.
 77    - `widget` The widget to place.
 78    - `spatial_map` The spatial map to check for available space.
 79
 80    **Returns:**
 81    - `bool` True if the widget was placed successfully, False otherwise.
 82
 83    Place a widget in a container, avoiding overlap with other widgets.
 84    Placement occurs virtually be checking if the widget fits in the container using a spatial map.
 85    The placing is done randomly within the container space and by checking the spatial map to avoid overlap with other widgets.
 86    Placement is attempted 100 times before giving up.
 87    """
 88    for _ in range(100):  # Try 100 times to find a spot
 89        x = random.randint(0, container.width - widget.get_width())
 90        y = random.randint(0, container.height - widget.get_height())
 91        if spatial_map.is_space_available(x, y, widget.get_width(), widget.get_height()):
 92            widget.set_pos(x, y)
 93            spatial_map.occupy_space(x, y, widget.get_width(), widget.get_height())
 94            return True
 95    return False  # Could not place the widget

224    def place_widget(self, widget, spatial_map: SpatialMap):
225        """
226        **Params:**
227        - `widget` The widget to place.
228        - `spatial_map` The spatial map to check for available space.
229
230        **Returns:**
231        - `bool` True if the widget was placed successfully, False otherwise.
232
233        Place a widget in a container, avoiding overlap with other widgets.
234        Placement occurs virtually be checking if the widget fits in the container using a spatial map.
235        The placing is done randomly within the container space and by checking the spatial map to avoid overlap with other widgets.
236        Placement is attempted 100 times before giving up.
237        """
238        max_attempts = 100  # Max attempts to find a suitable spot
239        for _ in range(max_attempts):
240            x = random.randint(0, self.width - widget.get_width())
241            y = random.randint(0, self.height - widget.get_height())
242            if spatial_map.is_space_available(x, y, widget.get_width(), widget.get_height()):
243                widget.set_pos(x, y)
244                spatial_map.occupy_space(x, y, widget.get_width(), widget.get_height())
245                return True
246        return False  # Could not place the widget
